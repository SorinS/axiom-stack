 24 axiom.SystemFile = function(path) {
 25    var BufferedReader            = java.io.BufferedReader;
 26    var File                      = java.io.File;
 27    var Reader                    = java.io.Reader;
 28    var Writer                    = java.io.Writer;
 29    var FileReader                = java.io.FileReader;
 30    var FileWriter                = java.io.FileWriter;
 31    var PrintWriter               = java.io.PrintWriter;
 32    var EOFException              = java.io.EOFException;
 33    var IOException               = java.io.IOException;
 34    var IllegalStateException     = java.lang.IllegalStateException;
 35    var IllegalArgumentException  = java.lang.IllegalArgumentException;
 36 
 37    var self = this;
 38    var file;
 39 
 40    try {
 41       if (arguments.length > 1)
 42          file = new File(path, arguments[1]);
 43       else
 44          file = new File(path);
 45    } catch (e) {
 46       throw(e);
 47    }
 48 
 49    var readerWriter;
 50    var atEOF = false;
 51    var lastLine = null;
 52 
 53    var setError = function(e) {
 54       this.lastError = e;
 55    };
 56 
 57    this.lastError = null;
 58 
 59    this.toString = function() {
 60       return file.toString();
 61    };
 62 
 63    this.getName = function() {
 64       var name = file.getName();
 65       return (name == null ? "" : name);
 66    };

 68    this.isOpened = function() {
 69       return (readerWriter != null);
 70    };
 71 
 72    this.open = function(mode) {
 73       if (self.isOpened()) {
 74          setError(new IllegalStateException("File already open"));
 75          return false;
 76       }
 77        try {
 78 	   if (mode && mode == 'w') {
 79 	       //open to write to the file
 80 	       readerWriter = new PrintWriter(new FileWriter(file));
 81 	   } else if (mode && mode == 'a') {
 82 	       //append anything written to the file
 83 	       readerWriter = new PrintWriter(new FileWriter(file, true));
 84 	   } else if (file.exists()) {
 85 	       //default to reading the file
 86                readerWriter = new BufferedReader(new FileReader(file));
 87 	   } else {
 88 	       setError("File does not exist and cannot be read.");
 89 	       return false;
 90 	   }
 91 	   return true;
 92        } catch (e) {
 93            setError(e);
 94            return false;
 95        }
 96        return;
 97    };

 99    this.exists = function() {
100       return file.exists();
101    };
102 
103    this.getParent = function() {
104       if (!file.getParent())
105          return null;
106       return new axiom.SystemFile(file.getParent());
107    };

109    this.readln = function() {
110       if (!self.isOpened()) {
111          setError(new IllegalStateException("File not opened"));
112          return null;
113       }
114       if (!(readerWriter instanceof BufferedReader)) {
115          setError(new IllegalStateException("File not opened for reading"));
116          return null;
117       }
118       if (atEOF) {
119          setError(new EOFException());
120          return null;
121       }
122       if (lastLine != null) {
123          var line = lastLine;
124          lastLine = null;
125          return line;
126       }
127       var reader = readerWriter;
128       // Here lastLine is null, return a new line
129       try {
130          var line = readerWriter.readLine();
131          if (line == null) {
132             atEOF = true;
133             setError(new EOFException());
134          }
135          return line;
136       } catch (e) {
137          setError(e);
138          return null;
139       }
140       return;
141    };
142 
143    this.write = function(what) {
144       if (!self.isOpened()) {
145          setError(new IllegalStateException("File not opened"));
146          return false;
147       }
148       if (!(readerWriter instanceof PrintWriter)) {
149          setError(new IllegalStateException("File not opened for writing"));
150          return false;
151       }
152       if (what != null) {
153          readerWriter.print(what.toString());
154       }
155       return true;
156    };
157 
158    this.writeln = function(what) {
159       if (self.write(what)) {
160          readerWriter.println();
161          return true;
162       }
163       return false;
164    };

166    this.isAbsolute = function() {
167       return file.isAbsolute();
168    };

170    this.remove = function() {
171       if (self.isOpened()) {
172          setError(new IllegalStateException("An openened file cannot be removed"));
173          return false;
174       }
175       return file["delete"]();
176    };

178    /*
179     * will list all files within a directory
180     * you may pass a RegExp Pattern to return just
181     * files matching this pattern
182     * @example var xmlFiles = dir.list(/.*\.xml/);
183     * @param RegExp pattern to test each file name against
184     * @return Array the list of file names
185     */
186    this.list = function(pattern) {
187       if (self.isOpened())
188          return null;
189       if (!file.isDirectory())
190          return null;
191       if (pattern) {
192          var fileList = file.list();
193          var result = [];
194          for (var i in fileList) {
195             if (pattern.test(fileList[i]))
196                result.push(fileList[i]);
197          }
198          return result;
199       }
200       return file.list();
201    };

203    this.flush = function() {
204       if (!self.isOpened()) {
205          setError(new IllegalStateException("File not opened"));
206          return false;
207       }
208       if (readerWriter instanceof Writer) {
209          try {
210             readerWriter.flush();
211          } catch (e) {
212            setError(e);
213            return false;
214          }
215       } else {
216          setError(new IllegalStateException("File not opened for write"));
217          return false; // not supported by reader
218       }
219       return true;
220    };

222    this.close = function() {
223       if (!self.isOpened())
224          return false;
225       try {
226          readerWriter.close();
227          readerWriter = null;
228          return true;
229       } catch (e) {
230          setError(e);
231          readerWriter = null;
232          return false;
233       }
234    };

241    this.error = function() {
242       if (this.lastError == null) {
243          return "";
244       } else {
245          var exceptionName = this.lastError.getClass().getName();
246          var l = exceptionName.lastIndexOf(".");
247          if (l > 0)
248             exceptionName = exceptionName.substring(l + 1);
249          return exceptionName + ": " + this.lastError.getMessage();
250       }
251    };

294    this.renameTo = function(toFile) {
295       if (toFile == null) {
296          setError(new IllegalArgumentException("Uninitialized target File object"));
297          return false;
298       }
299       if (self.isOpened()) {
300          setError(new IllegalStateException("An openened file cannot be renamed"));
301          return false;
302       }
303       if (toFile.isOpened()) {
304          setError(new IllegalStateException("You cannot rename to an openened file"));
305          return false;
306       }
307       return file.renameTo(new java.io.File(toFile.getAbsolutePath()));
308    };

310    this.eof = function() {
311       if (!self.isOpened()) {
312          setError(new IllegalStateException("File not opened"));
313          return true;
314       }
315       if (!(readerWriter instanceof BufferedReader)) {
316          setError(new IllegalStateException("File not opened for read"));
317          return true;
318       }
319       if (atEOF)
320          return true;
321       if (lastLine != null)
322          return false;
323       try {
324          lastLine = readerWriter.readLine();
325          if (lastLine == null)
326             atEOF = true;
327          return atEOF;
328       } catch (e) {
329          setError(e);
330          return true;
331       }
332    };

334    this.readAll = function() {
335       // Open the file for readAll
336       if (self.isOpened()) {
337          setError(new IllegalStateException("File already open"));
338          return null;
339       }
340       try {
341          if (file.exists()) {
342             readerWriter = new BufferedReader(new FileReader(file));
343          } else {
344             setError(new IllegalStateException("File does not exist"));
345             return null;
346          }
347          if (!file.isFile()) {
348             setError(new IllegalStateException("File is not a regular file"));
349             return null;
350          }
351 
352          // read content line by line to setup proper eol
353          var buffer = new java.lang.StringBuffer(file.length() * 1.10);
354          while (true) {
355             var line = readerWriter.readLine();
356             if (line == null)
357                break;
358             if (buffer.length() > 0)
359                buffer.append("\n");  // EcmaScript EOL
360             buffer.append(line);
361          }
362 
363          // Close the file
364          readerWriter.close();
365          readerWriter = null;
366          return buffer.toString();
367       } catch (e) {
368          readerWriter = null;
369          setError(e);
370          return null;
371       }
372    };

375    // this method removes a directory recursively

377    this.removeDirectory = function() {
378       if (!file.isDirectory())
379          return false;
380       var arr = file.list();
381       for (var i=0; i<arr.length; i++) {
382          var f = new axiom.SystemFile(file, arr[i]);
383          if (f.isDirectory())
384             f.removeDirectory();
385          else
386             f.remove();
387       }
388       file["delete"]();
389       return true;
390    };

392    /**
393     * recursivly lists all files below a given directory
394     * you may pass a RegExp Pattern to return just
395     * files matching this pattern
396     * @param RegExp pattern to test each file name against
397     * @returns Array the list of absolute file paths
398     */
399    this.listRecursive = function(pattern) {
400       if (!file.isDirectory())
401          return false;
402       if (!pattern || pattern.test(file.getName()))
403          var result = [file.getAbsolutePath()];
404       else
405          var result = [];
406       var arr = file.list();
407       for (var i=0; i<arr.length; i++) {
408          var f = new axiom.SystemFile(file, arr[i]);
409          if (f.isDirectory())
410             result = result.concat(f.listRecursive(pattern));
411          else if (!pattern || pattern.test(arr[i]))
412             result.push(f.getAbsolutePath());
413       }
414       return result;
415    }

417    /**
418     * function makes a copy of a file over partitions
419     * @param StringOrFile full path of the new file
420     */
421    this.hardCopy = function(dest) {
422       var inStream = new java.io.BufferedInputStream(
423          new java.io.FileInputStream(file)
424       );
425       var outStream = new java.io.BufferedOutputStream(
426          new java.io.FileOutputStream(dest)
427       );
428       var buffer = java.lang.reflect.Array.newInstance(
429          java.lang.Byte.TYPE, 4096
430       );
431       var bytesRead = 0;
432       while ((bytesRead = inStream.read(buffer, 0, buffer.length)) != -1) {
433          outStream.write(buffer, 0, bytesRead);
434       }
435       outStream.flush();
436       inStream.close();
437       outStream.close();
438       return true;
439    }

441    /**
442     * function moves a file to a new destination directory
443     * @param String full path of the new file
444     * @return Boolean true in case file could be moved, false otherwise
445     */
446    this.move = function(dest) {
447       // instead of using the standard File method renameTo()
448       // do a hardCopy and then remove the source file. This way
449       // file locking shouldn't be an issue
450       self.hardCopy(dest);
451       // remove the source file
452       file["delete"]();
453       return true;
454    }

456    /**
457     * returns file as ByteArray
458     * useful for passing it to a function instead of an request object
459     */
460    this.toByteArray = function() {
461       if (!this.exists())
462          return null;
463       var body = new java.io.ByteArrayOutputStream();
464       var stream = new java.io.BufferedInputStream(
465          new java.io.FileInputStream(this.getAbsolutePath())
466       );
467       var buf = java.lang.reflect.Array.newInstance(
468          java.lang.Byte.TYPE, 1024
469       );
470       var read;
471       while ((read = stream.read(buf)) > -1)
472          body.write(buf, 0, read);
473       stream.close();
474       return body.toByteArray();
475    };

488 /**
489  * Return the string content of the file. Caller should handle potential i/o exceptions.
490  */
491 axiom.SystemFile.readFromFile = function(filename){
492 	var reader;
493 	var lines = [];
494 	try{
495 		var f = new java.io.File(filename);
496 		if(!f.exists()) {
497 			return '';
498 		}
499 		reader = new java.io.BufferedReader(new java.io.FileReader(f));
500 
501 		var line;
502 		while((line = reader.readLine()) !== null){
503 			lines.push(line);
504 		}
505 	} finally{
506 		if(reader)
507 			reader.close();
508 		reader = null;
509 	}
510 	return lines.join('\n');
511 }

520 		writer.print(str);
521 	} finally{
522 		if(writer)
523 			writer.close();
524 		writer = null;
525 	}
526 }
