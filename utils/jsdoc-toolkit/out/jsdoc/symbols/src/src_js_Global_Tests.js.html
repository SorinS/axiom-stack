 20 axiom.Test = {};
 21 axiom.Test.globalSuite = function(){
 22 	var suite = { global: (global._test || {}) };
 23 	for each (var proto in app.getPrototypes()){
 24 		if (global[proto] && global[proto].prototype){
 25 			suite[proto] = (global[proto].prototype._test || {});
 26 		}
 27 	}
 28 	return suite;
 29 }

 31 axiom.Test.run = function(suite){
 32 	var results = { 
 33 		errors: 0,
 34 		failures: 0,
 35 		test_results: []
 36 	};
 37 
 38 	var start = new Date();
 39 	for(var proto in suite){
 40 		var tests = suite[proto];
 41 		axiom.Test.runSuite(tests, results, proto);
 42 	}
 43 	results.total_time = ((new Date()).getTime() - start.getTime() ) / 1000.0;
 44 	return results;
 45 }

 47 axiom.Test.runSuite = function(tests, results, proto){
 48 	var setup = (tests.setup || function(){});
 49 	var teardown = (tests.teardown || function(){});
 50 	for(var test in tests){
 51 		if(test.match(/^test/) && typeof tests[test] == 'function'){
 52 			setup.call(tests);
 53 			res.commit();
 54 			var test_result = { name: proto+'.'+test };
 55 			var test_start = new Date()
 56 			try {
 57 				tests[test].call(tests);
 58 			} catch(e) {
 59 				if(e instanceof AssertionFailedError){
 60 					results.failures += 1;
 61 					test_result.failure = e;
 62 				} else {
 63 					results.errors += 1;
 64 					test_result.error = e;
 65 				}
 66 			} finally {
 67 				test_result.time = ( (new Date()).getTime() - test_start.getTime() ) / 1000.0;
 68 			}
 69 			results.test_results.push(test_result);
 70 			res.commit();
 71 			teardown.call(tests);
 72 			res.commit();
 73 		} else if(typeof tests[test] == 'object'){
 74 			setup.call(tests);
 75 			res.commit();			
 76 			axiom.Test.runSuite(tests[test], results, proto+'.'+test)
 77 			teardown.call(tests);
 78 			res.commit();
 79 		}
 80 	}
 81 }
