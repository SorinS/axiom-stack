 17 function TAL(doc, data) {
 18     var tal = new Namespace('tal', 'http://axiomstack.com/tale');
 19     TAL.Scope.prototype = data;
 20     local_data = new TAL.Scope();
 21 	default xml namespace = doc.namespace('');
 22 	if(doc.namespace('tal')==tal) {
 23         TAL.TALE(doc, local_data, tal);
 24     } else {
 25         tal = new Namespace('tal', 'http://xml.zope.org/namespaces/tal');
 26         local_data['repeat']={};
 27         TAL.TAL(doc, local_data, data, tal);
 28     }
 29 
 30     doc.removeNamespace(tal);
 31     return doc;
 32 }
 33 TAL.namespace_transform = function(doc, ns_transform){
 34     for each(node in doc..*){
 35         for each(namespace in ns_transform){
 36             var to = namespace.to;
 37             var from = namespace.from;
 38             var transforms = node.@from::*;
 39             if(transforms.length() > 0){
 40                 for each(attr in transforms){
 41                     node.@to::[attr.name()] = attr.toString();
 42                     delete node.@from::[attr.name()];
 43                 }
 44             }
 45         }
 46     }
 47     return doc;
 48 }

 54 TAL.TALE = function (n, data, tal) {
 55         var tn;
 56         data.node=n;
 57         if((tn=n.@tal::repeat).length()) {
 58                 var r = TAL.terms(data, tn);
 59                 delete n.@tal::repeat;
 60                 var p=n.parent();
 61                 for (var l in r) {
 62                     var collection = r[l];
 63 					if(!collection) continue;
 64 					for (var k in collection){
 65                         TAL.Scope.prototype = data;
 66                         var repeat_data = new TAL.Scope();
 67                         repeat_data[l] = collection[k];
 68                         if(!repeat_data['repeat']) repeat_data['repeat'] = {}
 69                         repeat_data['repeat'][l] = new TAL.Repeat(k, collection.length);
 70                         p.insertChildBefore(n, n.copy());
 71                         TAL.TALE(p.*[n.childIndex()-1], repeat_data, tal);
 72                     }
 73                 }
 74                 delete p.*[n.childIndex()];
 75                 return;
 76         }
 77         if((tn=n.@tal::['var']).length()) {
 78                 TAL.Scope.prototype = data;
 79                 data = new TAL.Scope();
 80                 var r = TAL.terms(data, tn);
 81                 for(var k in r) data[k]=r[k];
 82                 delete n.@tal::['var'];
 83         }
 84         if((tn=n.@tal::eval).length()) {
 85                 TAL.func(data, tn);
 86                 delete n.@tal::eval;
 87         }
 88         if((tn=n.@tal::['if']).length()) {
 89                 var r = TAL.func(data, tn);
 90                  if(r) {
 91                         delete n.@tal::['if'];
 92                 } else {
 93                         delete n.parent().*[n.childIndex()];
 94                         return;
 95                 }
 96         }
 97         if((tn=n.@tal::['repeat-content']).length()) {
 98             delete n.@tal::['repeat-content'];
 99             n.replace('*', <kill tal:omit="true" tal:repeat={tn} xmlns:tal={tal.uri}>{n.*}</kill>);
100         }
101        if((tn=n.@tal::['content-if']).length()) {
102             var r = TAL.func(data, tn);
103             if(r) {
104                 n.replace('*', r);
105             } else {
106                 delete n.parent().*[n.childIndex()];
107             }
108             delete n.@tal::['content-if'];
109         }
110         if((tn=n.@tal::attr).length()) {
111                 var r = TAL.terms(data, tn);
112                 for(var k in r)
113                     if(r[k]){
114                         var ns_match = k.match(/(\w+):(\w+)/);
115                         if(ns_match){
116                             var ns = n.namespace([ns_match[1]]);
117                             n.@ns::[ns_match[2]] = r[k];
118                         }
119                         else
120                             n.@[k]=r[k];
121                     }
122                 delete n.@tal::attr;
123         }
124         if((tn=n.@tal::content).length()) {
125                 n.replace('*', TAL.func(data, tn));
126                 delete n.@tal::content;
127 
128         }
129         if((tn=n.@tal::replace).length()) {
130                 n.parent().replace(n.childIndex(), TAL.func(data, tn));
131                 return;
132         }
133         if((tn=n.@tal::text).length()) {
134                 var r=new RegExp('\\'+tn+'\\{([^}]+)\\}|\\'+tn+'(\\w+)','g');
135                 delete n.@tal::text;
136                 for each(var t in n.text()){
137                     n.replace(t.childIndex(),
138                               new XMLList(t.toXMLString().replace(r,
139 	                function(m,m1,m2){return TAL.func(data, m1||m2)})));
140             }
141         }
142         if((tn=n.@tal::omit).length()) {
143             if(TAL.func(data, tn)) {
144 				for each(var child in n.*){
145 					if((child..@tal::*).length()>0) TAL.TALE(child, data, tal);
146 				}
147 				n.parent().replace(n.childIndex(), n.*);
148 				return;
149 			} else{
150 				delete n.@tal::omit;
151 			}
152         }
153     for each (var child in n.*) {
154         if((child..@tal::*).length()>0) TAL.TALE(child, data, tal);
155     }
156 }
157 TAL.semisplit = function (s) {
158     var r=[], v=s.split(';'), p;
159     while((p=v.shift())!=null) {
160         if(p.length==0) { r.push(r.pop()+';'+v.shift()); }
161         else { r.push(p); }
162     }
163     return r;
164 }
165 TAL.RE_define = /^ *(global )?(local )? *([^ ]+) +(.+)$/;
166 TAL.RE_nameExpr = /^ *([^ ]+) +(.*)$/;
167 TAL.RE_trim = /^ *(.*?) *$/;
168 TAL.Repeat = function (i, c) { this.index = i; this.count = c; }
169 TAL.Repeat.prototype.number = function () { return parseInt(this.index,10) + 1 }
170 TAL.Repeat.prototype.even = function () { return ((this.index % 2)==0)?true:false; }
171 TAL.Repeat.prototype.odd = function () { return ((this.index % 2)==1)?true:false; }
172 TAL.Repeat.prototype.start = function () { return this.index==0 }
173 TAL.Repeat.prototype.end = function () { return this.index==(this.count-1) }
174 TAL.stringExpr = function(data, str) {
175         var path_re=/\$\{([^}]+)\}|\$([^$ \/]+)/g, lastlast=0, parts=[], match, index;
176 while((match=path_re.exec(str))!=null) {
177     index = match.index;
178     if(index==0 || str.charAt(index-1)!='$') {
179         parts.push(str.slice(lastlast, index).replace('$$', '$'));
180         parts.push(TAL.getPathExpr(data, match[1] || match[2])[1]);
181     } else {
182         parts.push(str.slice(lastlast, path_re.lastIndex).replace('$$', '$'));
183     }
184     lastlast = path_re.lastIndex;
185 }
186 parts.push(str.slice(lastlast).replace('$$','$'));
187 return parts.join('');
188 }

190 /*
191  * pathExpr - look up the value of a path variable in the current scope.
192  *                returns a two-element array --> [bool, exprValue], where the
193  *                first element is true if the path variable was defined, and the
194  *                second element is the value itself of the variable
195  */
196 TAL.getPathExpr = function(data, path) {
197     var original_path = path;
198     var path = path.split("/");
199     var value = data;
200     var old_value;
201     for(var j=0; j< path.length; j++) {
202         old_value = value;
203         if(!value) return [false, value];
204         value = value[path[j]];
205         switch(typeof value) {
206         case "undefined":
207             // support backwards compatible clown scoping
208             var func = data['this'][path[j]]
209             if(func && typeof func == "function"){
210                 return [true, func.call(data['this'], data)];
211             }
212             return [false, value];
213             break;
214         case "function":
215             value = value.call(old_value);
216             if(!value) return [false, value];
217             break;
218         }
219     }
220     return [true, value];
221 }
222 TAL.expr = function (data, value) {
223     // look for conditional expressions
224     var parts = value.split("|?");
225     if(parts.length > 1){
226         var conds = parts[1].split("|:");
227         var result = TAL.expr(data, parts[0]);
228         if(!result)
229             return TAL.expr(data, conds[1]);
230         else
231             return TAL.expr(data, conds[0]);
232 
233     }
234     // deal with pipes
235     var terms = value.split("|");
236     for(var i=0; i<terms.length; i++) {
237         var term = terms[i];
238         if(term.indexOf("string:")==0 || term.indexOf("javascript:")==0) {
239             terms.splice(i, terms.slice(i).join('|'));
240             break;
241         }
242     }
243     for(var i=0; i < terms.length; i++) {
244         var exists=false, nocall=false, string=false, not=false, javascript=false;
245         var term = terms[i].replace(TAL.RE_trim, '$1');
246 
247         // look for prefixes
248         while(true) {
249             var term7 = term.slice(0, 7);
250             switch(term7) {
251             case 'exists:': exists=true; term=term.slice(7); continue;
252             case 'nocall:': nocall=true; term=term.slice(7); continue;
253             case 'string:': string=true; term=term.slice(7); continue;
254             }
255             if(term.slice(0, 4)=="not:") { not=true; term=term.slice(4); continue; }
256             if(term.slice(0, 11)=="javascript:") { javascript=true; term=term.slice(11); continue; }
257             break;
258         }
259         if(string) {
260             return TAL.stringExpr(data, term);
261         } else if(javascript) {
262             return (new Function('path','return ' + term)).call(data['this'],function (path){return TAL.getPathExpr(data, TAL.stringExpr(data, path))[1]});
263         }
264         result = TAL.getPathExpr(data, TAL.stringExpr(data, term));
265         if(!result[1] && i+1 < terms.length ){
266             continue;
267         }
268         if(exists) return not ? !result[0] : result[0];
269         return not ? !result[1] : result[1];
270     }
271 }

273 TAL.TAL = function(n, data, globalData, tal) {
274     var tn;
275     if((tn=n.@tal::define).length()) {
276             TAL.Scope.prototype = data;
277             data = new TAL.Scope();
278             for each (var t in TAL.semisplit(tn.toString())) {
279                 var glnx = t.match(TAL.RE_define);
280                 if(glnx[1]) {
281                         globalData[glnx[3]] = TAL.expr(data, glnx[4]);
282                 } else {
283                         data[glnx[3]] = TAL.expr(data, glnx[4]);
284                 }
285             }
286             delete n.@tal::define;
287     }
288     if((tn=n.@tal::condition).length()) {
289             var r = TAL.expr(data, tn.toString());
290             if(r) {
291                 delete n.@tal::['condition'];
292             } else {
293                 delete n.parent().*[n.childIndex()];
294                 return;
295             }
296     }
297     if((tn=n.@tal::['repeat-content']).length()) {
298             delete n.@tal::['repeat-content'];
299             var nn=new XML('<kill tal:repeat="'+tn+'" tal:omit-tag="string:1" xmlns:tal="'+tal.uri+'">'+n.toXMLString()+'</kill>');
300             n.replace('*', nn);
301     }
302     if((tn=n.@tal::repeat).length()) {
303         var nx = tn.toString().match(TAL.RE_nameExpr);
304         var v = TAL.expr(data, nx[2]), p = n.parent();
305         delete n.@tal::repeat;
306         var c = v?v.length:0;
307 
308         var repeat_lambda = function(i){
309             TAL.Scope.prototype = data;
310             repeat_data = new TAL.Scope();
311             repeat_data['repeat'][nx[1]] = new TAL.Repeat(i, c);
312             repeat_data[nx[1]] = v[i];
313             p.insertChildBefore(n, n.copy());
314             TAL.TAL(p.*[n.childIndex()-1], repeat_data, globalData, tal);
315         };
316 
317         if(v instanceof MultiValue){
318             // XXX workaround since multivalues aren't iteritable at the moment
319             for(var i=0; i<c; i++)
320                 repeat_lambda(i);
321         } else {
322             for(var i in v)
323                 repeat_lambda(i);
324         }
325         delete p.*[n.childIndex()];
326         return;
327     }
328     if((tn=n.@tal::content).length()) {
329             n.replace('*', TAL.expr(data, tn.toString()));
330             delete n.@tal::content;
331     }
332     if((tn=n.@tal::replace).length()) {
333         n.parent().replace(n.childIndex(), TAL.expr(data, tn.toString()));
334             return;
335     }
336     if((tn=n.@tal::attributes).length()) {
337             for each (var t in TAL.semisplit(tn.toString())) {
338                 var nx = t.match(TAL.RE_nameExpr);
339                 var expr_result = TAL.expr(data, nx[2]);
340                 if(expr_result)
341                     n.@[nx[1]] = expr_result;
342             }
343             delete n.@tal::attributes;
344     }
345     if((tn=n.@tal::text).length()) {
346             var r=new RegExp('\\'+tn+'\\{([^}]+)\\}|\\'+tn+'(\\w+)','g');
347             delete n.@tal::text;
348             for each(var t in n.text())
349                 n.replace(t.childIndex(),
350                           new XMLList(t.toXMLString().replace(r,
351 	            function(m,m1,m2){return TAL.expr(data, m1 || m2);})));
352     }
353     for each (var c in n.*) {
354         TAL.TAL(c, data, globalData, tal);
355     }
356     if((tn=n.@tal::['omit-tag']).length()) {
357             if(TAL.expr(data, tn.toString())) n.parent().replace(n.childIndex(), n.*.copy());
358 				else delete n.@tal::['omit-tag'];
359     }
