 18 Date.ONESECOND    = 1000;
 19 Date.ONEMINUTE    = 60 * Date.ONESECOND;
 20 Date.ONEHOUR      = 60 * Date.ONEMINUTE;
 21 Date.ONEDAY       = 24 * Date.ONEHOUR;
 22 Date.ONEWEEK      =  7 * Date.ONEDAY;

 25 /**
 26  * format a Date to a string
 27  * @param String Format pattern
 28  * @param Object Java Locale Object (optional)
 29  * @param Object Java TimeZone Object (optional)
 30  * @return String formatted Date
 31  */
 32 Date.prototype.format = function (format, locale, timezone) {
 33     if (!format)
 34         return this.toString();
 35     var sdf = locale ? new java.text.SimpleDateFormat(format, locale)
 36                      : new java.text.SimpleDateFormat(format);
 37     if (timezone && timezone != sdf.getTimeZone())
 38         sdf.setTimeZone(timezone);
 39     return sdf.format(this);
 40 };

 43 Date.prototype.dateValue = function() {
 44 	var t = this.getTime() / 86400000;
 45 	t = Packages.axiom.objectmodel.dom.LuceneDataFormatter.roundUpDouble(t);
 46     return new java.text.DecimalFormat("000000").format(0 + t);
 47 };

 50 Date.prototype.timeValue = function() {
 51 	var t = this.getTime() / 1000;
 52 	t = Packages.axiom.objectmodel.dom.LuceneDataFormatter.roundUpDouble(t);
 53     return new java.text.DecimalFormat("00000000000").format(0 + t);
 54 };

 57 Date.prototype.timestampValue = function() {
 58 	var t = Packages.axiom.objectmodel.dom.LuceneDataFormatter.roundUpDouble(this.getTime());
 59     return new java.text.DecimalFormat("00000000000000").format(0 + t);
 60 };
