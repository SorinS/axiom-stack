 25 /**
 26  * constructor function for Zip Objects
 27  * @param either
 28  *      - (Object) a File object
 29  *      - (Object) an instance of Helma.File
 30  *      - (Object) an instance of java.io.File
 31  *      - (String) the path to the zip file
 32  * @constructor
 33  */
 34 axiom.Zip = function (file) {
 35 
 36     /**
 37      * private function that extracts the data of a file
 38      * in a .zip archive. If a destination Path is given it
 39      * writes the extracted data directly to disk using the
 40      * name of the ZipEntry Object, otherwise it returns the
 41      * byte array containing the extracted data
 42      * @param Object jAva.util.zip.ZipFile Object
 43      * @param Object java.util.zip.ZipEntry Object to extract
 44      * @param String destination path to extract ZipEntry Object to
 45      * @return Object ByteArray containing the data of the ZipEntry
 46      */
 47     var extractEntry = function(zFile, entry, destPath) {
 48         var size = entry.getSize();
 49         if (entry.isDirectory() || size <= 0)
 50             return null;
 51 
 52         var zInStream = new java.io.BufferedInputStream(zFile.getInputStream(entry));
 53         var buf = new java.lang.reflect.Array.newInstance(java.lang.Byte.TYPE, size);
 54         zInStream.read(buf, 0, size);
 55         zInStream.close();
 56 
 57         if (!destPath) {
 58             // no filesystem destination given, so return
 59             // the byte array containing the extracted data
 60             return buf;
 61         }
 62         // extract the file to the given path
 63         var dest = new java.io.File(destPath, entry.getName());
 64         if (entry.isDirectory())
 65             dest.mkdirs();
 66         else if (buf) {
 67             if (!dest.getParentFile().exists())
 68                 dest.getParentFile().mkdirs();
 69             try {
 70                 var outStream = new java.io.BufferedOutputStream(new java.io.FileOutputStream(dest));
 71                 outStream.write(buf, 0, size);
 72             } finally {
 73                 outStream.close();
 74             }
 75         }
 76         return null;
 77     };
 78 
 79     /**
 80      * private function for adding a single file to the .zip archive
 81      * @param Object java.util.zip.ZipOutputStream
 82      * @param Object instance of java.io.File representing the file to be added
 83      * @param Int compression-level (0-9)
 84      * @param String path of the directory in the archive to which the
 85      *                    file should be added (optional)
 86      */
 87     var addFile = function(zOutStream, f, level, pathPrefix) {
 88         var fInStream = new java.io.BufferedInputStream(new java.io.FileInputStream(f));
 89         buf = new java.lang.reflect.Array.newInstance(java.lang.Byte.TYPE, f.length());
 90         fInStream.read(buf, 0, f.length());
 91 
 92         var name = new java.lang.StringBuffer();
 93         if (pathPrefix) {
 94             // append clean pathPrefix to name buffer
 95             var st = new java.util.StringTokenizer(pathPrefix, "\\/");
 96             while (st.hasMoreTokens()) {
 97                 name.append(st.nextToken());
 98                 name.append("/");
 99             }
100         }
101         name.append(f.getName());
102         var entry = new java.util.zip.ZipEntry(name.toString());
103         entry.setSize(f.length());
104         entry.setTime(f.lastModified());
105         zOutStream.setLevel(level);
106         zOutStream.putNextEntry(entry);
107         zOutStream.write(buf, 0, buf.length);
108         zOutStream.closeEntry();
109         fInStream.close();
110         return true;
111     };
112 
113     /**
114      * private function that constructs an instance
115      * of java.io.File based on a JS File or Helma.File object
116      * @param Object either a string or an instance of java.io.File, File or Helma.File
117      * @return Object instance of java.io.File
118      */
119     var evalFile = function(f) {
120         if (f instanceof java.io.File)
121             return f;
122         var result;
123         if (typeof f == "string")
124             result = new java.io.File(f);
125         else
126             result = new java.io.File(f.getAbsolutePath());
127         if (!result.exists())
128             throw("Error creating Zip Object: File '" + f + "' doesn't exist.");
129         return result;
130     };
131 
132     /**
133      * returns an array containing the entries of a .zip file as objects
134      * (see Entry for description)
135      * @param Object File object representing the .zip file on disk
136      * @return Object result object
137      */
138     this.list = function() {
139         var result = new axiom.Zip.Content();
140         var zFile = new java.util.zip.ZipFile(file);
141         var entries = zFile.entries();
142         while (entries.hasMoreElements())
143             result.add(new axiom.Zip.Entry(entries.nextElement()));
144         zFile.close();
145         return result;
146     };
147 
148     /**
149      * extracts a single file from a .zip archive
150      * if a destination path is given it directly writes
151      * the extracted file to disk
152      * @param Object File object representing the .zip file on disk
153      * @param String Name of the file to extract
154      * @param String (optional) destination path to write file to
155      * @return Object JS Object (see Entry for description)
156      */
157     this.extract = function(name, destPath) {
158         var zFile = new java.util.zip.ZipFile(file);
159         var entry = zFile.getEntry(name);
160         if (!entry)
161             return null;
162         var result = new axiom.Zip.Entry(entry);
163         result.data = extractEntry(zFile, entry, destPath);
164         zFile.close();
165         return result;
166     };
167 
168     /**
169      * extracts all files in a .zip archive
170      * if a destination path is given it directly writes
171      * the extracted files to disk (preserves directory structure
172      * of .zip archive if existing!)
173      * @param String (optional) destination path to write file to
174      * @return Object Array containing JS objects for each entry
175      *                     (see Entry for description)
176      */
177     this.extractAll = function(destPath) {
178         var result = new axiom.Zip.Content();
179         var zFile = new java.util.zip.ZipFile(file);
180         var entries = zFile.entries();
181         while (entries.hasMoreElements()) {
182             var entry = entries.nextElement();
183             var e = new axiom.Zip.Entry(entry);
184             e.data = extractEntry(zFile, entry, destPath);
185             result.add(e);
186         }
187         zFile.close();
188         return result;
189     };
190 
191     /**
192      * adds a single file or a whole directory (recursive!) to the .zip archive
193      * @param either
194      *          - (Object) a File object
195      *          - (Object) an instance of java.io.File
196      *          - (String) the path to the file that should be added
197      * @param Int Level to use for compression (default: 9 = best compression)
198      * @param String name of the directory in the archive into which the
199      *                    file should be added (optional)
200      * @return Boolean true
201      */
202     this.add = function (f, level, pathPrefix) {
203         var f = evalFile(f);
204 
205         // evaluate arguments
206         if (arguments.length == 2) {
207             if (typeof arguments[1] == "string") {
208                 pathPrefix = arguments[1];
209                 level = 9;
210             } else {
211                 level = parseInt(arguments[1], 10);
212                 pathPrefix = null;
213             }
214         } else if (level == null || isNaN(level)) {
215             level = 9;
216         }
217         // only levels between 0 and 9 are allowed
218         level = Math.max(0, Math.min(9, level));
219 
220         if (f.isDirectory()) {
221             // add a whole directory to the zip file (recursive!)
222             var files = (new axiom.File(f.getAbsolutePath())).listRecursive();
223             for (var i in files) {
224                 var fAdd = new java.io.File(files[i]);
225                 if (!fAdd.isDirectory()) {
226                     var p = fAdd.getPath().substring(f.getAbsolutePath().length, fAdd.getParent().length);
227                     if (pathPrefix)
228                         p = pathPrefix + p;
229                     addFile(zOutStream, fAdd, level, p);
230                 }
231             }
232         } else
233             addFile(zOutStream, f, level, pathPrefix);
234         return true;
235     };
236 
237     /**
238      * adds a new entry to the zip file
239      * @param Object byte[] containing the data to add
240      * @param String name of the file to add
241      * @param Int compression level (0-9, default: 9)
242      * @return Boolean true
243      */
244     this.addData = function(buf, name, level) {
245         var entry = new java.util.zip.ZipEntry(name);
246         entry.setSize(buf.length);
247         entry.setTime(new Date());
248         if (level == null || isNaN(level))
249             zOutStream.setLevel(9);
250         else
251             zOutStream.setLevel(Math.max(0, Math.min(9, parseInt(level, 10))));
252         zOutStream.putNextEntry(entry);
253         zOutStream.write(buf, 0, buf.length);
254         zOutStream.closeEntry();
255         return true;
256     };
257 
258     /**
259      * closes the ZipOutputStream
260      */
261     this.close = function() {
262         zOutStream.close();
263         return;
264     };
265 
266     /**
267      * returns the binary data of the zip file
268      * @return Object ByteArray containing the binary data of the zip file
269      */
270     this.getData = function() {
271         return bOutStream.toByteArray();
272      };
273 
274     /**
275      * saves the archive by closing the output stream
276      * @param String path (including the name) to save the zip file to
277      * @return Boolean true
278      */
279     this.save = function(dest) {
280         if (!dest)
281             throw new Error("no destination for ZipFile given");
282         // first of all, close the ZipOutputStream
283         zOutStream.close();
284         var destFile = new java.io.File(dest);
285         try {
286             var outStream = new java.io.FileOutputStream(destFile);
287             bOutStream.writeTo(outStream);
288         } finally {
289             outStream.close();
290         }
291         return true;
292     };
293 
294     this.toString = function() {
295         if (file)
296             return "[Zip Object " + file.getAbsolutePath() + "]";
297         else
298             return "[Zip Object]";
299     };

301     /**
302      * constructor body
303      */
304     var bOutStream = new java.io.ByteArrayOutputStream();
305     var zOutStream = new java.util.zip.ZipOutputStream(bOutStream);
306     if (file) {
307         file = evalFile(file);
308     }
309 
310     for (var i in this)
311         this.dontEnum(i);
312 
313     return this;
314 }

316 /**
317  * constructor for Content Objects
318  */
319 axiom.Zip.Content = function() {
320     this.toc = [];
321     this.files = {};
322 
323     /**
324      * adds a Zip Entry object to the table of contents
325      * and the files collection
326      * @param Object instance of axiom.Zip.Entry
327      */
328     this.add = function(entry) {
329         // add the file to the table of contents array
330         this.toc.push(entry);
331         // plus add it to the files tree
332         var re = /[\\\/]/;
333         var arr = entry.name.split(re);
334         var cnt = 0;
335         var curr = this.files;
336         var propName;
337         while (cnt < arr.length-1) {
338             propName = arr[cnt++];
339             if (!curr[propName])
340                 curr[propName] = new Object();
341             curr = curr[propName];
342         }
343         curr[arr[cnt]] = entry;
344         return;
345     };
346 
347 	for (var i in this)
348 		this.dontEnum(i);
349 
350     return this;
351 };

354 /**
355  * constructor for Entry objects holding the meta-data of a zip entry:
356  *     .name (String) name of the entry
357  *     .size (Int) decompressed size of the entry in bytes
358  *     .time (Date) last modification timestamp of the entry or null
359  *     .isDirectory (Boolean) true in case entry is a directory, false otherwise
360  *     .data (ByteArray) ByteArray containing the data of the entry
361  * @param Object java.util.zip.ZipEntry Object
362  */
363 axiom.Zip.Entry = function(entry) {
364     this.name = entry.getName();
365     this.size = entry.getSize();
366     this.time = entry.getTime() ? new Date(entry.getTime()) : null;
367     this.isDirectory = entry.isDirectory();
368     this.data = null;
369     for (var i in this)
370         this.dontEnum(i);
371     return this;
372 };

375 /**
376  * extract all files in a ByteArray passed as argument
377  * and return them as result Array
378  * @param Object ByteArray containing the data of the .zip File
379  * @return Object instance of axiom.Zip.Content
380  */
381 axiom.Zip.extractData = function(zipData) {
382     var zInStream = new java.util.zip.ZipInputStream(new java.io.ByteArrayInputStream(zipData));
383     var result = new axiom.Zip.Content();
384 
385     var entry;
386     while ((entry = zInStream.getNextEntry()) != null) {
387         var eParam = new axiom.Zip.Entry(entry);
388         if (eParam.isDirectory)
389             continue;
390         if (eParam.size == -1)
391             eParam.size = 16384;
392         var bos = new java.io.ByteArrayOutputStream(eParam.size);
393         var buf = java.lang.reflect.Array.newInstance(java.lang.Byte.TYPE, 8192);
394         var count;
395         while ((count = zInStream.read(buf)) != -1)
396             bos.write(buf, 0, count);
397         eParam.data = bos.toByteArray();
398         eParam.size = bos.size();
399         result.add(eParam);
400     }
401     zInStream.close();
402     return result;
403 };
