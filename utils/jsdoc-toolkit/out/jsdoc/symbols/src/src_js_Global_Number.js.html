 18 Number.prototype.smallIntValue = function() {
 19 	return new java.text.DecimalFormat("0000").format(0 + this);
 20 }
 21 
 22 Number.prototype.smallFloatValue = function() {
 23 	return new java.text.DecimalFormat("0000.0000").format(0 + this);
 24 }
